CREATE TABLE students ( student_id INT PRIMARY KEY AUTO_INCREMENT, name VARCHAR(100) NOT NULL, email VARCHAR(100) UNIQUE NOT NULL ); CREATE TABLE courses ( course_id INT PRIMARY KEY , course_name VARCHAR(100) NOT NULL, UNIQUE (course_id) ); CREATE TABLE enrollments ( enrollment_id INT PRIMARY KEY AUTO_INCREMENT, student_id INT NOT NULL, course_id INT NOT NULL, FOREIGN KEY (student_id) REFERENCES students(student_id), FOREIGN KEY (course_id) REFERENCES courses(course_id), UNIQUE (student_id, course_id) );
